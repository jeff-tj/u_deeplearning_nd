Preprocess the Data¶

Since the project's main focus is on building the GANs, we'll preprocess the data for you. The values of the MNIST and CelebA dataset will be in the range of -0.5 to 0.5 of 28x28 dimensional images. The CelebA images will be cropped to remove parts of the image that don't include a face, then resized down to 28x28.

The MNIST images are black and white images with a single color channel while the CelebA images have 3 color channels (RGB color channel).

def discriminator(images, reuse=False):
    """
    Create the discriminator network
    :param images: Tensor of input image(s)
    :param reuse: Boolean if the weights should be reused
    :return: Tuple of (tensor output of the discriminator, tensor logits of the discriminator)
    """
    # TODO: Implement Function
    with tf.variable_scope("discriminator", reuse=reuse):
        # To Do:
        # [] Improve tf.variable_scope understanding
        # [] Build a better network
        init_depth = 64
        alpha = 0.2
        # For images we will use a conv setup like in the exercises
        # Images are initially 28x28xinput_channels
        x = tf.layers.conv2d(images, filters=init_depth, kernel_size=5, 
                             strides=2, padding="same")
        x = tf.maximum(alpha*x, x)
        # Now (14 x 14 x init_depth)
        
        x = tf.layers.conv2d(x, filters=init_depth*2, kernel_size=5, 
                             strides=2, padding="same")
        x = tf.layers.batch_normalization(x, training=True)
        x = tf.maximum(alpha*x, x)
        # Now (7 x 7 x init_depth*2)
        
        x = tf.layers.conv2d(x, filters=init_depth*4, kernel_size=5, 
                             strides=2, padding="same")
        x = tf.layers.batch_normalization(x, training=True)
        x = tf.maximum(alpha*x, x)
        # Now (4 x 4 x init_depth*4)
        
        # Flatten
        x = tf.layers.flatten(x)
        logits =  tf.layers.dense(x, 1, activation=None)
        out = tf.sigmoid(logits)
        
        return out, logits

Generator¶

Implement generator to generate an image using z. This function should be able to reuse the variables in the neural network. Use tf.variable_scope with a scope name of "generator" to allow the variables to be reused. The function should return the generated 28 x 28 x out_channel_dim images.

def generator(z, out_channel_dim, is_train=True):
    """
    Create the generator network
    :param z: Input z
    :param out_channel_dim: The number of channels in the output image
    :param is_train: Boolean if generator is being used for training
    :return: The tensor output of the generator
    """
    # TODO: Implement Function
    alpha = 0.2
    # Follow up
    # [] Reuse - should be false in training? True in inference
    with tf.variable_scope("generator", reuse=(not is_train)):
        init_depth = 512
        x = tf.layers.dense(z, 4*4*init_depth)
        x = tf.reshape(x, (-1, 4, 4, init_depth)) # Tensor dim: (batch, h, w, d)
        x = tf.layers.batch_normalization(x, training=is_train)
        x = tf.maximum(alpha*x, x)
        # Now (4 x 4 x init_depth)
        
        # Convolutional transpose
        x = tf.layers.conv2d_transpose(x, filters=init_depth//2, kernel_size=5, 
                                       strides=2, padding="same")
        x = tf.layers.batch_normalization(x, training=is_train)
        x = tf.maximum(alpha*x, x)
        # Now (8 x 8 x init_depth//2)
        
        x = tf.layers.conv2d_transpose(x, filters=init_depth//4, kernel_size=5, 
                                       strides=2, padding="same")
        x = tf.layers.batch_normalization(x, training=is_train)
        x = tf.maximum(alpha*x, x)
        # Now (16 x 16 x init_depth//4)
        
        x = tf.layers.conv2d_transpose(x, filters=out_channel_dim, kernel_size=5, 
                                       strides=2, padding="same")
        # Now (32 x 32 x out_channel_dim)
        
        # Need to resize image
        # [] Alternative to crop tf.image.resize_image_with_crop_or_pad (but loss)
        logits = tf.image.resize_images(x, size=[28, 28], method=tf.image.ResizeMethod.BILINEAR)
        out = tf.tanh(logits)
    return out

def model_loss(input_real, input_z, out_channel_dim):
    """
    Get the loss for the discriminator and generator
    :param input_real: Images from the real dataset
    :param input_z: Z input
    :param out_channel_dim: The number of channels in the output image
    :return: A tuple of (discriminator loss, generator loss)
    """
    # TODO: Implement Function
    # Generate some images
    input_genr = generator(input_z, out_channel_dim, is_train=True)
    
    # Test the discrominator - with output and logits for real and generated
    real_out, real_logits = discriminator(input_real, reuse=False)
    genr_out, genr_logits = discriminator(input_genr, reuse=True)
    
    # Discriminator loss functions
    smooth = 0.9 # Implement a smoothing like in the lessons
    d_loss_real = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=real_logits, 
                                                labels=smooth*tf.ones_like(real_out)))
    d_loss_genr = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=genr_logits, 
                                                labels=tf.zeros_like(genr_out)))
    d_loss = d_loss_real + d_loss_genr
    # Gene
    g_loss = tf.reduce_mean(
        tf.nn.sigmoid_cross_entropy_with_logits(logits=genr_logits, 
                                                labels=tf.ones_like(genr_out)))
    return d_loss, g_loss

def model_opt(d_loss, g_loss, learning_rate, beta1):
    """
    Get optimization operations
    :param d_loss: Discriminator loss Tensor
    :param g_loss: Generator loss Tensor
    :param learning_rate: Learning Rate Placeholder
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :return: A tuple of (discriminator training operation, generator training operation)
    """
    # TODO: Implement Function
    # Get the variables for each part
    t_vars = tf.trainable_variables()
    d_vars = [var for var in t_vars if var.name.startswith("discriminator")]
    g_vars = [var for var in t_vars if var.name.startswith("generator")]
    
    # With batch normalisation, need to use tf.control_dependencies
    with tf.control_dependencies(tf.get_collection(tf.GraphKeys.UPDATE_OPS)):
        d_opt = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(d_loss, var_list=d_vars)
        g_opt = tf.train.AdamOptimizer(learning_rate, beta1=beta1).minimize(g_loss, var_list=g_vars)
    
    return d_opt, g_opt

def show_generator_output(sess, n_images, input_z, out_channel_dim, image_mode):
    """
    Show example output for the generator

def train(epoch_count, batch_size, z_dim, learning_rate, beta1, get_batches, data_shape, data_image_mode):
    """
    Train the GAN
    :param epoch_count: Number of epochs
    :param batch_size: Batch Size
    :param z_dim: Z dimension
    :param learning_rate: Learning Rate
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :param get_batches: Function to get batches
    :param data_shape: Shape of the data
    :param data_image_mode: The image mode to use for images ("RGB" or "L")
    """
    # TODO: Build Model
    import matplotlib
    if data_image_mode == "RGB":
        img_ch = 3
    else:
        img_ch = 1
        
    # Some variables to control
    losses = []
    graph_every = 5
    show_every = 100
    step = 0
    
    # Model
    #tf.reset_default_graph()
    
    # [] Do we need to assign the learning rate variable
    input_real, input_z, learn_rate = model_inputs(data_shape[1], data_shape[2], img_ch, z_dim)
    d_loss, g_loss = model_loss(input_real, input_z, img_ch)
    d_opt, g_opt = model_opt(d_loss, g_loss, learning_rate, beta1)
    
    # saver to save checkpoints
    saver = tf.train.Saver()
    
    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        for epoch_i in range(epoch_count):
            for batch_images in get_batches(batch_size):
                # TODO: Train Model
                step += 1
                # Generate batches (rescale and generate noise)
                batch_images *= 2
                batch_z = np.random.uniform(1, -1, size=(batch_size, z_dim))
                # Run optimizers
                _ = sess.run(d_opt, feed_dict={input_real: batch_images,
                                               input_z: batch_z,
                                               learn_rate: learning_rate})
                _ = sess.run(g_opt, feed_dict={input_real: batch_images,
                                               input_z: batch_z,
                                               learn_rate: learning_rate})
                # Log results for graph
                if step % graph_every == 0:
                    train_loss_d = d_loss.eval({input_real: batch_images,
                                                input_z: batch_z,
                                                learn_rate: learning_rate})
                    train_loss_g = g_loss.eval({input_real: batch_images,
                                                input_z: batch_z,
                                                learn_rate: learning_rate})
                    losses.append((train_loss_d, train_loss_g))
                # Print results
                if step % show_every == 0:
                    train_loss_d = d_loss.eval({input_real: batch_images,
                                                input_z: batch_z,
                                                learn_rate: learning_rate})
                    train_loss_g = g_loss.eval({input_real: batch_images,
                                                input_z: batch_z,
                                                learn_rate: learning_rate})
                    print("Epoch: {}/{}...".format(epoch_i+1, epoch_count), 
                          "Discrimnator Loss: {:.4f}...".format(train_loss_d),
                          "Generator Loss: {:.4f}".format(train_loss_g))
                    show_generator_output(sess, 20, input_z, img_ch, data_image_mode)        
        saver.save(sess, "./checkpoints/genarator.ckpt")
        # Print final results
        train_loss_d = d_loss.eval({input_real: batch_images,
                                    input_z: batch_z,
                                    learn_rate: learning_rate})
        train_loss_g = g_loss.eval({input_real: batch_images,
                                    input_z: batch_z,
                                    learn_rate: learning_rate})
        print("Epoch: {}/{}...".format(epoch_i+1, epoch_count), 
              "Discrimnator Loss: {:.4f}...".format(train_loss_d),
              "Generator Loss: {:.4f}".format(train_loss_g))
        show_generator_output(sess, 20, input_z, img_ch, data_image_mode) 
        # Graph losses
        fig, ax = pyplot.subplots()
        losses = np.array(losses)
        pyplot.plot(losses.T[0], label='Discriminator', alpha=0.5)
        pyplot.plot(losses.T[1], label='Generator', alpha=0.5)
        pyplot.title("Training Losses")
        pyplot.legend()
